LIBAOM command line changelog

from 02/01/2020 --------------------

aomenc -o <bitstream> --verbose --passes=1 --cpu-used=0 --threads=1 --ivf --i420 --profile=0 --width=<width> --height=<height> --input-bit-depth=<bitdept> --fps=<fps> --kf-max-dist=<kf interval> --kf-min-dist=<kf interval> --lag-in-frames=25 --aq-mode=0 --end-usage=q --auto-alt-ref=1 --frame-parallel=0 --tile-columns=0 --test-decode=fatal --cq-level=<qp> <reconstructed YUV>

<kf interval> = 4 seconds (e.g. 120 frames for 30 fps, 96 frames for 24 fps)

<qp> = {20, 32, 43, 55, 63}
